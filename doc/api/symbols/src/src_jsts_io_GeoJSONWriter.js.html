  7 (function() {
  8     /**
  9      * Writes the GeoJSON representation of a {@link Geometry}. The
 10      * The GeoJSON format is defined <A
 11      * HREF="http://geojson.org/geojson-spec.html">here</A>.
 12      * <p>
 13      * The <code>GeoJSONWriter</code> outputs coordinates rounded to the precision
 14      * model. Only the maximum number of decimal places necessary to represent the
 15      * ordinates to the required precision will be output.
 16      * <p>
 17      *
 18      * @see WKTReader
 19      * @constructor
 20      */
 21     jsts.io.GeoJSONWriter = function() {
 22       this.parser = new jsts.io.GeoJSONParser(this.geometryFactory);
 23     };
 24 
 25     /**
 26      * Converts a <code>Geometry</code> to its GeoJSON representation.
 27      *
 28      * @param {jsts.geom.Geometry}
 29      *          geometry a <code>Geometry</code> to process.
 30      * @return {Object} The GeoJSON representation of the Geometry.
 31      */
 32 
 33     jsts.io.GeoJSONWriter.prototype.write = function(geometry) {
 34       var geoJson = this.parser.write(geometry);
 35 
 36       return geoJson;
 37     };
 38 })();
