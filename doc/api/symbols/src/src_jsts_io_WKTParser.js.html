 15 /**
 16  * Class for reading and writing Well-Known Text.
 17  *
 18  * NOTE: Adapted from OpenLayers 2.11 implementation.
 19  */

 21 /**
 22  * Create a new parser for WKT
 23  *
 24  * @param {}
 25  *          geometryFactory
 26  * @return An instance of WKTParser.
 27  */
 28 jsts.io.WKTParser = function(geometryFactory) {
 29   this.geometryFactory = geometryFactory || new jsts.geom.GeometryFactory();
 30 
 31   this.regExes = {
 32     'typeStr': /^\s*(\w+)\s*\(\s*(.*)\s*\)\s*$/,
 33     'emptyTypeStr': /^\s*(\w+)\s*EMPTY\s*$/,
 34     'spaces': /\s+/,
 35     'parenComma': /\)\s*,\s*\(/,
 36     'doubleParenComma': /\)\s*\)\s*,\s*\(\s*\(/, // can't use {2} here
 37     'trimParens': /^\s*\(?(.*?)\)?\s*$/
 38   };
 39 };
 40 
 41 
 42 /**
 43  * Deserialize a WKT string and return a geometry. Supports WKT for POINT,
 44  * MULTIPOINT, LINESTRING, LINEARRING, MULTILINESTRING, POLYGON, MULTIPOLYGON,
 45  * and GEOMETRYCOLLECTION.
 46  *
 47  * @param {String}
 48  *          wkt A WKT string.
 49  * @return {jsts.geom.Geometry} A geometry instance.
 50  */
 51 jsts.io.WKTParser.prototype.read = function(wkt) {
 52   var geometry, type, str;
 53   wkt = wkt.replace(/[\n\r]/g, ' ');
 54   var matches = this.regExes.typeStr.exec(wkt);
 55   if (wkt.search('EMPTY') !== -1) {
 56     matches = this.regExes.emptyTypeStr.exec(wkt);
 57     matches[2] = undefined;
 58   }
 59   if (matches) {
 60     type = matches[1].toLowerCase();
 61     str = matches[2];
 62     if (this.parse[type]) {
 63       geometry = this.parse[type].apply(this, [str]);
 64     }
 65   }
 66 
 67   if (geometry === undefined)
 68     throw new Error('Could not parse WKT ' + wkt);
 69 
 70   return geometry;
 71 };
 72 
 73 /**
 74  * Serialize a geometry into a WKT string.
 75  *
 76  * @param {jsts.geom.Geometry}
 77  *          geometry A feature or array of features.
 78  * @return {String} The WKT string representation of the input geometries.
 79  */
 80 jsts.io.WKTParser.prototype.write = function(geometry) {
 81   return this.extractGeometry(geometry);
 82 };
 83 
 84 /**
 85  * Entry point to construct the WKT for a single Geometry object.
 86  *
 87  * @param {jsts.geom.Geometry}
 88  *          geometry
 89  *
 90  * @return {String} A WKT string of representing the geometry.
 91  */
 92 jsts.io.WKTParser.prototype.extractGeometry = function(geometry) {
 93   var type = geometry.CLASS_NAME.split('.')[2].toLowerCase();
 94   if (!this.extract[type]) {
 95     return null;
 96   }
 97   var wktType = type.toUpperCase();
 98   var data;
 99   if (geometry.isEmpty()) {
100     data = wktType + ' EMPTY';
101   } else {
102     data = wktType + '(' + this.extract[type].apply(this, [geometry]) + ')';
103   }
104   return data;
105 };
106 
107 /**
108  * Object with properties corresponding to the geometry types. Property values
109  * are functions that do the actual data extraction.
110  */
111 jsts.io.WKTParser.prototype.extract = {
112   'coordinate': function(coordinate) {
113     return coordinate.x + ' ' + coordinate.y;
114   },
115 
116   /**
117    * Return a space delimited string of point coordinates.
118    *
119    * @param {jsts.geom.Point}
120    *          point
121    * @return {String} A string of coordinates representing the point.
122    */
123   'point': function(point) {
124     return point.coordinate.x + ' ' + point.coordinate.y;
125   },
126 
127   /**
128    * Return a comma delimited string of point coordinates from a multipoint.
129    *
130    * @param {jsts.geom.MultiPoint>}
131    *          multipoint
132    * @return {String} A string of point coordinate strings representing the
133    *         multipoint.
134    */
135   'multipoint': function(multipoint) {
136     var array = [];
137     for (var i = 0, len = multipoint.geometries.length; i < len; ++i) {
138       array.push('(' +
139           this.extract.point.apply(this, [multipoint.geometries[i]]) + ')');
140     }
141     return array.join(',');
142   },
143 
144   /**
145    * Return a comma delimited string of point coordinates from a line.
146    *
147    * @param {jsts.geom.LineString>}
148    *          linestring
149    * @return {String} A string of point coordinate strings representing the
150    *         linestring.
151    */
152   'linestring': function(linestring) {
153     var array = [];
154     for (var i = 0, len = linestring.points.length; i < len; ++i) {
155       array.push(this.extract.coordinate.apply(this, [linestring.points[i]]));
156     }
157     return array.join(',');
158   },
159 
160   /**
161    * Return a comma delimited string of linestring strings from a
162    * multilinestring.
163    *
164    * @param {jsts.geom.MultiLineString>}
165    *          multilinestring
166    * @return {String} A string of of linestring strings representing the
167    *         multilinestring.
168    */
169   'multilinestring': function(multilinestring) {
170     var array = [];
171     for (var i = 0, len = multilinestring.geometries.length; i < len; ++i) {
172       array.push('(' +
173           this.extract.linestring.apply(this, [multilinestring.geometries[i]]) +
174           ')');
175     }
176     return array.join(',');
177   },
178 
179   /**
180    * Return a comma delimited string of linear ring arrays from a polygon.
181    *
182    * @param {jsts.geom.Polygon>}
183    *          polygon
184    * @return {String} An array of linear ring arrays representing the polygon.
185    */
186   'polygon': function(polygon) {
187     var array = [];
188     array
189         .push('(' + this.extract.linestring.apply(this, [polygon.shell]) + ')');
190     for (var i = 0, len = polygon.holes.length; i < len; ++i) {
191       array.push('(' + this.extract.linestring.apply(this, [polygon.holes[i]]) +
192           ')');
193     }
194     return array.join(',');
195   },
196 
197   /**
198    * Return an array of polygon arrays from a multipolygon.
199    *
200    * @param {jsts.geom.MultiPolygon>}
201    *          multipolygon
202    * @return {String} An array of polygon arrays representing the multipolygon.
203    */
204   'multipolygon': function(multipolygon) {
205     var array = [];
206     for (var i = 0, len = multipolygon.geometries.length; i < len; ++i) {
207       array.push('(' +
208           this.extract.polygon.apply(this, [multipolygon.geometries[i]]) + ')');
209     }
210     return array.join(',');
211   },
212 
213   /**
214    * Return the WKT portion between 'GEOMETRYCOLLECTION(' and ')' for an
215    * geometrycollection.
216    *
217    * @param {jsts.geom.GeometryCollection>}
218    *          collection
219    * @return {String} internal WKT representation of the collection.
220    */
221   'geometrycollection': function(collection) {
222     var array = [];
223     for (var i = 0, len = collection.geometries.length; i < len; ++i) {
224       array.push(this.extractGeometry.apply(this, [collection.geometries[i]]));
225     }
226     return array.join(',');
227   }
228 
229 };

231 /**
232  * Object with properties corresponding to the geometry types. Property values
233  * are functions that do the actual parsing.
234  */
235 jsts.io.WKTParser.prototype.parse = {
236   /**
237    * Return point geometry given a point WKT fragment.
238    *
239    * @param {String}
240    *          str A WKT fragment representing the point.
241    * @return {jsts.geom.Point} A point geometry.
242    * @private
243    */
244   'point': function(str) {
245     if (str === undefined) {
246       return this.geometryFactory.createPoint(null);
247     }
248 
249     var coords = str.trim().split(this.regExes.spaces);
250     return this.geometryFactory.createPoint(new jsts.geom.Coordinate(coords[0],
251         coords[1]));
252   },
253 
254   /**
255    * Return a multipoint geometry given a multipoint WKT fragment.
256    *
257    * @param {String}
258    *          A WKT fragment representing the multipoint.
259    * @return {jsts.geom.Point} A multipoint feature.
260    * @private
261    */
262   'multipoint': function(str) {
263     if (str === undefined) {
264       return this.geometryFactory.createMultiPoint(null);
265     }
266 
267     var point;
268     var points = str.trim().split(',');
269     var components = [];
270     for (var i = 0, len = points.length; i < len; ++i) {
271       point = points[i].replace(this.regExes.trimParens, '$1');
272       components.push(this.parse.point.apply(this, [point]));
273     }
274     return this.geometryFactory.createMultiPoint(components);
275   },
276 
277   /**
278    * Return a linestring geometry given a linestring WKT fragment.
279    *
280    * @param {String}
281    *          A WKT fragment representing the linestring.
282    * @return {jsts.geom.LineString} A linestring geometry.
283    * @private
284    */
285   'linestring': function(str) {
286     if (str === undefined) {
287       return this.geometryFactory.createLineString(null);
288     }
289 
290     var points = str.trim().split(',');
291     var components = [];
292     var coords;
293     for (var i = 0, len = points.length; i < len; ++i) {
294       coords = points[i].trim().split(this.regExes.spaces);
295       components.push(new jsts.geom.Coordinate(coords[0], coords[1]));
296     }
297     return this.geometryFactory.createLineString(components);
298   },
299 
300   /**
301    * Return a linearring geometry given a linearring WKT fragment.
302    *
303    * @param {String}
304    *          A WKT fragment representing the linearring.
305    * @return {jsts.geom.LinearRing} A linearring geometry.
306    * @private
307    */
308   'linearring': function(str) {
309     if (str === undefined) {
310       return this.geometryFactory.createLinearRing(null);
311     }
312 
313     var points = str.trim().split(',');
314     var components = [];
315     var coords;
316     for (var i = 0, len = points.length; i < len; ++i) {
317       coords = points[i].trim().split(this.regExes.spaces);
318       components.push(new jsts.geom.Coordinate(coords[0], coords[1]));
319     }
320     return this.geometryFactory.createLinearRing(components);
321   },
322 
323   /**
324    * Return a multilinestring geometry given a multilinestring WKT fragment.
325    *
326    * @param {String}
327    *          A WKT fragment representing the multilinestring.
328    * @return {jsts.geom.MultiLineString} A multilinestring geometry.
329    * @private
330    */
331   'multilinestring': function(str) {
332     if (str === undefined) {
333       return this.geometryFactory.createMultiLineString(null);
334     }
335 
336     var line;
337     var lines = str.trim().split(this.regExes.parenComma);
338     var components = [];
339     for (var i = 0, len = lines.length; i < len; ++i) {
340       line = lines[i].replace(this.regExes.trimParens, '$1');
341       components.push(this.parse.linestring.apply(this, [line]));
342     }
343     return this.geometryFactory.createMultiLineString(components);
344   },
345 
346   /**
347    * Return a polygon geometry given a polygon WKT fragment.
348    *
349    * @param {String}
350    *          A WKT fragment representing the polygon.
351    * @return {jsts.geom.Polygon} A polygon geometry.
352    * @private
353    */
354   'polygon': function(str) {
355     if (str === undefined) {
356       return this.geometryFactory.createPolygon(null);
357     }
358 
359     var ring, linestring, linearring;
360     var rings = str.trim().split(this.regExes.parenComma);
361     var shell;
362     var holes = [];
363     for (var i = 0, len = rings.length; i < len; ++i) {
364       ring = rings[i].replace(this.regExes.trimParens, '$1');
365       linestring = this.parse.linestring.apply(this, [ring]);
366       linearring = this.geometryFactory.createLinearRing(linestring.points);
367       if (i === 0) {
368         shell = linearring;
369       } else {
370         holes.push(linearring);
371       }
372 
373     }
374     return this.geometryFactory.createPolygon(shell, holes);
375   },
376 
377   /**
378    * Return a multipolygon geometry given a multipolygon WKT fragment.
379    *
380    * @param {String}
381    *          A WKT fragment representing the multipolygon.
382    * @return {jsts.geom.MultiPolygon} A multipolygon geometry.
383    * @private
384    */
385   'multipolygon': function(str) {
386     if (str === undefined) {
387       return this.geometryFactory.createMultiPolygon(null);
388     }
389 
390     var polygon;
391     var polygons = str.trim().split(this.regExes.doubleParenComma);
392     var components = [];
393     for (var i = 0, len = polygons.length; i < len; ++i) {
394       polygon = polygons[i].replace(this.regExes.trimParens, '$1');
395       components.push(this.parse.polygon.apply(this, [polygon]));
396     }
397     return this.geometryFactory.createMultiPolygon(components);
398   },
399 
400   /**
401    * Return a geometrycollection given a geometrycollection WKT fragment.
402    *
403    * @param {String}
404    *          A WKT fragment representing the geometrycollection.
405    * @return {jsts.geom.GeometryCollection}
406    * @private
407    */
408   'geometrycollection': function(str) {
409     if (str === undefined) {
410       return this.geometryFactory.createGeometryCollection(null);
411     }
412 
413     // separate components of the collection with |
414     str = str.replace(/,\s*([A-Za-z])/g, '|$1');
415     var wktArray = str.trim().split('|');
416     var components = [];
417     for (var i = 0, len = wktArray.length; i < len; ++i) {
418       components.push(jsts.io.WKTParser.prototype.read.apply(this,
419           [wktArray[i]]));
420     }
421     return this.geometryFactory.createGeometryCollection(components);
422   }
423 
424 };
